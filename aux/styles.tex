%%% Нумерация и названия таблиц и рисунков %%%

\def\capfigure{figure}
%\def\captable{table}
\long\def\@makecaption#1#2
{%
  \vskip\abovecaptionskip
  \ifx\@captype\capfigure
      \centering #1~--~#2 \par
  \else
      #1~--~#2 \par
  \fi
  \vskip\belowcaptionskip
}

\captionsetup[figure]{font=small}
